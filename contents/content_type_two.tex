\newcommand{\insertTitle}[1]{
    \ifnum#1=1
        \section{第一章标题}
    \fi
    \ifnum#1=7
        \section{第二章标题}
    \fi
    \ifnum#1=23
        \section{第三章标题}
    \fi
    \ifnum#1=56
        \section{第四章标题}
    \fi
    \ifnum#1=78
        \section{第五章标题}
    \fi
}
